index = 0

for fullname in dataset_fullnames:
    dataset = pd.read_csv(fullname)
    percent_ages0to14 = dataset["percent_ages0to14"]
    percent_15_or_older = 100 - percent_ages0to14

    plt.hist(percent_15_or_older, color='red', alpha=0.7)
